#bin/bash
#pkg install git php -y
git clone https://github.com/ridwan12782/PemantauJaringan.git
cat >start-PemantauJaringan <<EOL
#bin/bash
if pgrep "php" >/dev/null 2>&1 ; then
killall php
fi
cd PemantauJaringanV3
php -S 0.0.0.0:8080
EOL
cat >update-PemantauJaringan <<EOL
#bin/bash
mkdir ~/backup-PemantauJaringan
cp ~/PemantauJaringanV3/include/config.php ~/backup-PemantauJaringan/config.php
cp -rf ~/PemantauJaringanV3/voucher/. ~/backup-PemantauJaringan/voucher/
cp -rf ~/PemantauJaringanV3/img/. ~/backup-PemantauJaringan/img/
rm -rf PemantauJaringanV3
rm ~/backup-PemantauJaringan/voucher/print.php
rm ~/backup-PemantauJaringan/voucher/printbt.php
git clone https://github.com/ridwan12782/PemantauJaringan.git
cp ~/backup-PemantauJaringanV3/config.php ~/PemantauJaringanV3/include/config.php
cp -rf ~/backup-PemantauJaringanV3/voucher/. ~/PemantauJaringanV3/voucher/
cp -rf ~/backup-PemantauJaringanV3/img/. ~/PemantauJaringanV3/img/
rm -rf ~/backup-PemantauJaringan
./start-PemantauJaringan
EOL
cat >remove-PemantauJaringan <<EOL
#bin/bash
rm -rf PemantauJaringanV3
EOL
cat >.bashrc <<EOL	
~/start-PemantauJaringan
EOL
chmod +x start-PemantauJaringan
chmod +x update-PemantauJaringan
chmod +x remove-PemantauJaringan
./start-PemantauJaringan
